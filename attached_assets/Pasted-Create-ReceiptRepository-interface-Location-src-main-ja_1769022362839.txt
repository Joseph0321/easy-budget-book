package com.budget.easybook.receipt.repository;

import com.budget.easybook.receipt.entity.Receipt;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

@Repository
public interface ReceiptRepository extends JpaRepository<Receipt, Long> {
    
    /**
     * Find all receipts for a user with pagination
     * @param userId user ID
     * @param pageable pagination info
     * @return page of receipts
     */
    @Query("SELECT r FROM Receipt r WHERE r.user.userId = :userId ORDER BY r.processedAt DESC")
    Page<Receipt> findByUserId(@Param("userId") Long userId, Pageable pageable);
    
    /**
     * Find receipt by ID and verify ownership
     * @param receiptId receipt ID
     * @param userId user ID
     * @return Optional containing receipt if found and owned by user
     */
    @Query("SELECT r FROM Receipt r WHERE r.receiptId = :receiptId AND r.user.userId = :userId")
    Optional<Receipt> findByIdAndUserId(@Param("receiptId") Long receiptId, @Param("userId") Long userId);
    
    /**
     * Find recent receipts for a user (last N receipts)
     * @param userId user ID
     * @return list of recent receipts
     */
    @Query("SELECT r FROM Receipt r WHERE r.user.userId = :userId " +
           "ORDER BY r.processedAt DESC LIMIT 10")
    List<Receipt> findRecentByUserId(@Param("userId") Long userId);
    
    /**
     * Find receipts within date range
     * @param userId user ID
     * @param startDate start date
     * @param endDate end date
     * @param pageable pagination info
     * @return page of receipts
     */
    @Query("SELECT r FROM Receipt r WHERE r.user.userId = :userId " +
           "AND r.processedAt BETWEEN :startDate AND :endDate ORDER BY r.processedAt DESC")
    Page<Receipt> findByUserIdAndDateBetween(@Param("userId") Long userId,
                                              @Param("startDate") LocalDateTime startDate,
                                              @Param("endDate") LocalDateTime endDate,
                                              Pageable pageable);
    
    /**
     * Find receipts without linked expenses
     * @param userId user ID
     * @return list of unlinked receipts
     */
    @Query("SELECT r FROM Receipt r WHERE r.user.userId = :userId " +
           "AND NOT EXISTS (SELECT e FROM Expense e WHERE e.receipt.receiptId = r.receiptId) " +
           "ORDER BY r.processedAt DESC")
    List<Receipt> findUnlinkedByUserId(@Param("userId") Long userId);
    
    /**
     * Count receipts for a user
     * @param userId user ID
     * @return count
     */
    @Query("SELECT COUNT(r) FROM Receipt r WHERE r.user.userId = :userId")
    long countByUserId(@Param("userId") Long userId);
    
    /**
     * Count receipts processed in date range
     * @param userId user ID
     * @param startDate start date
     * @param endDate end date
     * @return count
     */
    @Query("SELECT COUNT(r) FROM Receipt r WHERE r.user.userId = :userId " +
           "AND r.processedAt BETWEEN :startDate AND :endDate")
    long countByUserIdAndDateBetween(@Param("userId") Long userId,
                                      @Param("startDate") LocalDateTime startDate,
                                      @Param("endDate") LocalDateTime endDate);
}

Important Notes:
1. processedAt is LocalDateTime (includes time)
2. LIMIT in JPQL for recent receipts
3. NOT EXISTS subquery for finding unlinked receipts
4. Useful for receipt management and cleanup
