// src/services/api.js
import axios from 'axios';

const api = axios.create({
  baseURL: process.env.VUE_APP_API_BASE_URL || 'http://localhost:8080/api',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
});

// Request/Response 인터셉터
api.interceptors.request.use(
  config => {
    // 추후 토큰 추가
    return config;
  },
  error => Promise.reject(error)
);

api.interceptors.response.use(
  response => response,
  error => {
    // 에러 처리
    console.error('API Error:', error);
    return Promise.reject(error);
  }
);

export default api;

// src/services/transactionService.js
import api from './api';

export default {
  // 거래 목록 조회
  getTransactions(params) {
    return api.get('/transactions', { params });
  },
  
  // 기간별 조회
  getTransactionsByPeriod(startDate, endDate) {
    return api.get('/transactions/period', {
      params: { start: startDate, end: endDate }
    });
  },
  
  // 거래 생성
  createTransaction(transaction) {
    return api.post('/transactions', transaction);
  },
  
  // 거래 수정
  updateTransaction(id, transaction) {
    return api.put(`/transactions/${id}`, transaction);
  },
  
  // 거래 삭제
  deleteTransaction(id) {
    return api.delete(`/transactions/${id}`);
  },
  
  // 월별 통계
  getMonthlySummary(year, month) {
    return api.get(`/transactions/monthly/${year}/${month}/summary`);
  },
  
  // Excel Export
  exportToExcel(startDate, endDate) {
    return api.get('/transactions/export/excel', {
      params: { start: startDate, end: endDate },
      responseType: 'blob'
    });
  },
  
  // Excel Import
  importFromExcel(file) {
    const formData = new FormData();
    formData.append('file', file);
    return api.post('/transactions/import/excel', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }
    });
  }
};
